library(readxl)
filter_RQ3 <- read_excel("Dataset/cbug.xlsx")
View(filter_RQ3)
layout(matrix(1:2,nr=1,nc=4),heights=c(1,1,1,1),widths=c(8,0,1,2))
par(cex.axis=1.5)
boxplot(filter_RQ3$COMMITS_ISSUES, horizontal = TRUE, ylim=c(0,150),boxwex=.8, cex=1.5)
par(cex.lab=1.5)
title(xlab="Commits per Bug")
text(boxplot.stats(filter_RQ3$COMMITS_ISSUES)$stats, labels = format(round(boxplot.stats(filter_RQ3$COMMITS_ISSUES)$stats,0),nsmall = 0), y = 0.75, cex=1.5)
stripchart(filter_RQ3$COMMITS_ISSUES[filter_RQ3$COMMITS_ISSUES>=400],pch=1,cex=1.5,xlim=c(400,1000))
